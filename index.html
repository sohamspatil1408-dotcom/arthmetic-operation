// Simple Arithmetic Program using Switch Case let a = 10; let b = 5; let operator = "+"; // Change operator to +, -, *, / let result; switch (operator) { case "+": result = a + b; break; case "-": result = a - b; break; case "*": result = a * b; break; case "/": result = a / b; break; default: result = "Invalid Operator"; } console.log("Result = " + result);
